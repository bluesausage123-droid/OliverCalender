        <p class="hint-text">例如：去醫院、買菜、找老張</p>
        <input type="text" id="tName" placeholder="請點這裡輸入內容">
        
        <button class="btn btn-voice" id="voiceBtn" onclick="startVoice()">🎤 用說的也可以</button>

        <hr style="margin: 30px 0; border: 1px solid #eee;">

        <label class="step-label">第二步：什麼時候提醒？</label>
        <input type="datetime-local" id="tTime">
        
        <button class="btn btn-save" onclick="saveTask()">💾 幫我記下來</button>
    </div>

    <div id="taskList"></div>
</div>

<script>
    // 導引標記：初始化預設時間
    window.onload = () => {
        if ("Notification" in window) Notification.requestPermission();
        
        // 自動填入「現在」的時間，長輩只需要改一下就好
        const now = new Date();
        now.setMinutes(now.getMinutes() - now.getTimezoneOffset());
        document.getElementById('tTime').value = now.toISOString().slice(0, 16);
        
        render();
        setInterval(checkReminders, 10000); // 每10秒檢查一次
    };

    // 導引標記：儲存邏輯 (自訂時間)
    function saveTask() {
        const name = document.getElementById('tName').value;
        const time = document.getElementById('tTime').value;

        if (!name) { alert("請告訴我要做什麼事喔！"); return; }
        
        const tasks = JSON.parse(localStorage.getItem('mars_tasks') || '[]');
        tasks.push({ id: Date.now(), name, time, done: false });
        localStorage.setItem('mars_tasks', JSON.stringify(tasks));
        
        document.getElementById('tName').value = "";
        render();
        alert("記好了！時間到我會大聲叫您。");
    }

    function render() {
        const listDiv = document.getElementById('taskList');
        const tasks = JSON.parse(localStorage.getItem('mars_tasks') || '[]');
        listDiv.innerHTML = tasks.sort((a,b)=>new Date(a.time)-new Date(b.time)).map(t => `
            <div class="item">
                <div class="item-info">
                    <b>${t.name}</b>
                    <span>⏰ ${t.time.replace('T', ' ')}</span>
                </div>
                <button class="btn-del" onclick="deleteTask(${t.id})">刪除</button>
            </div>
        `).join('');
    }

    function deleteTask(id) {
        let tasks = JSON.parse(localStorage.getItem('mars_tasks') || '[]');
        localStorage.setItem('mars_tasks', JSON.stringify(tasks.filter(t=>t.id!==id)));
        render();
    }

    // 導引標記：提醒觸發
    function checkReminders() {
        let tasks = JSON.parse(localStorage.getItem('mars_tasks') || '[]');
        const now = new Date();
        let updated = false;

        tasks.forEach(t => {
            const alarmTime = new Date(t.time);
            if (alarmTime <= now && !t.done) {
                fire(t.name);
                t.done = true;
                updated = true;
            }
        });
        if (updated) {
            localStorage.setItem('mars_tasks', JSON.stringify(tasks));
            render();
        }
    }

    function fire(name) {
        if (Notification.permission === "granted") {
            new Notification("⏰ 火星提醒：時間到！", { body: name });
        }
        const utter = new SpeechSynthesisUtterance("時間到囉，記得去做 " + name);
        utter.lang = 'zh-TW';
        window.speechSynthesis.speak(utter);
        if (navigator.vibrate) navigator.vibrate([500, 200, 500, 200, 500]);
    }

    // 導引標記：語音修復邏輯
    function startVoice() {
        const btn = document.getElementById('voiceBtn');
        const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
        
        if (!SpeechRecognition) {
            alert("抱歉，這台平板的瀏覽器不支援語音。建議使用 Chrome 瀏覽器。");
            return;
        }

        const recognition = new SpeechRecognition();
        recognition.lang = 'zh-TW';
        
        recognition.onstart = () => {
            btn.innerText = "🎤 正在聽您說話...";
            btn.classList.add('listening');
        };

        recognition.onerror = (event) => {
            console.error(event);
            btn.innerText = "🎤 用說的也可以";
            btn.classList.remove('listening');
            alert("語音辨識出錯了，請確認是否有開啟麥克風權限。");
        };

        recognition.onresult = (event) => {
            const text = event.results[0][0].transcript;
            document.getElementById('tName').value = text;
            btn.innerText = "🎤 用說的也可以";
            btn.classList.remove('listening');
        };

        recognition.onend = () => {
            btn.innerText = "🎤 用說的也可以";
            btn.classList.remove('listening');
        };

        recognition.start();
    }
</script>
